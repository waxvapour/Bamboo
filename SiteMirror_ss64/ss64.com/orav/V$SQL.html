Examples

SQL > Select substr(sql_text,0,80),
decode(command_type,2,'INSERT'),
executions
From V$SQL
Where command_type = 2
Order By executions;

SQL > Select substr(sql_text,0,80),
decode(command_type,3,'SELECT'),
executions
From V$SQL
Where command_type = 3
Order By executions;
